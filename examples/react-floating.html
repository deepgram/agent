Floating FAB React

FAB + overlay panel built on useDeepgramAgent. Click the mic button bottom-right.

Code

import { useState } from 'react';
import { useDeepgramAgent } from '@deepgram/agent-react';

function FloatingAgent() {
  const [open, setOpen] = useState(false);
  const { state, start, stop, conversation } =
    useDeepgramAgent({ config: { /* ... */ } });

  return (
    <>
      <button className="fab" onClick={() => setOpen(o => !o)}>🎙</button>
      {open && <Panel state={state} start={start} stop={stop}
                      conversation={conversation} />}
    </>
  );
}
