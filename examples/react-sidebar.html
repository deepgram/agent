Sidebar Layout React

Custom UI built directly on the useDeepgramAgent hook.

Code

import { useDeepgramAgent } from '@deepgram/agent-react';

function MySidebar() {
  const { state, conversation, start, stop, sendUserMessage } =
    useDeepgramAgent({
      config: {
        auth: { tokenFactory: () => fetch('/api/token')
          .then(r => r.json()).then(d => d.token) },
        agent: { /* ... */ },
      },
    });

  return (
    <div>
      <p>Status: {state}</p>
      <button onClick={start}>Start</button>
      {conversation.map(m => <p key={m.id}>{m.content}</p>)}
    </div>
  );
}
