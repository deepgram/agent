Deepgram Orb React UI

The signature Deepgram hoop animation. Three states: idle, listening, talking. Optional input/output volume for audio reactivity.

Usage

import { Orb } from '@deepgram/agent-react-ui';

// State-only
<Orb state={"listening"} />

// With auto volume (getter functions, sampled per frame)
<Orb state={"talking"}
  getOutputVolume={() => player.getOutputVolume()}
  getInputVolume={() => mic.getInputVolume()} />

// With manual volume (direct values)
<Orb state={"talking"} outputVolume={0.6} inputVolume={0.2} />

// Custom colors
<Orb state={"listening"} colors={["#6366f1", "#ec4899"]} />
