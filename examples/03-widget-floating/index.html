Floating FAB TS Source

Mic FAB opens an overlay panel. Change options to hot-reload.

Code

import { init } from '@deepgram/agents-widget';

const teardown = init({
  tokenFactory: () => fetch('/api/token').then(r => r.json()).then(d => d.token),
  agent: { /* ... */ },
  layout: 'floating',
  placement: 'bottom-right',
  colorScheme: 'dark',
});

// Unmount: teardown();
