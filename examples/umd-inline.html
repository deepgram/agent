Inline Layout Bundled UMD

Widget mounted into a host container via plain <script> tag — no bundler, no npm.

Code

<div id="agent-container" style="height:600px"></div>

<script src="https://cdn.deepgram.com/agent-widget/latest/widget.umd.js"></script>
<script>
  DeepgramAgent.init({
    tokenFactory: () => fetch('/api/token')
      .then(r => r.json()).then(d => d.token),
    agent: { /* ... */ },
    layout: 'inline',
    containerId: 'agent-container',
  });
</script>
