Console AI Bundled UMD

Deepgram Console AI assistant with navigation skills, tool calls, and context-aware actions.

Code

<script src="https://cdn.deepgram.com/agent-widget/latest/widget.umd.js"></script>
<script>
  DeepgramAgent.init({
    buttonId: 'dg-ask-ai-btn',
    skills: [
      {
        id: 'nav-dashboard',
        name: 'Go to Dashboard',
        description: 'Navigate to the project dashboard',
        category: 'navigation',
        risk: 'safe',
        parameters: [],
        execute: async (_params, ctx) => {
          ctx.navigate(`/project/${ctx.projectId}/dashboard`);
          return { success: true, message: 'Navigated to Dashboard.' };
        },
      },
      // ... 30+ skills for keys, billing, team, usage, etc.
    ],
    systemPrompt: 'You are a helpful assistant for the Deepgram Console...',
  });
</script>
